<script>
// 🔑 PUT YOUR FIREBASE CONFIG HERE
const firebaseConfig = {
 apiKey:"YOUR_API_KEY",
 authDomain:"YOUR_PROJECT.firebaseapp.com",
 databaseURL:"https://YOUR_PROJECT.firebaseio.com",
 projectId:"YOUR_PROJECT"
};

firebase.initializeApp(firebaseConfig);
const db = firebase.database();

let speeds = [];

function submitData(){
 let speed = parseFloat(document.getElementById('speed').value);
 let area = document.getElementById('area').value || "Unknown";
 if(!speed) return;

 db.ref('reports').push({
   speed: speed,
   area: area,
   time: Date.now()
 });
}

function draw(){
 let c = document.getElementById('chart');
 let ctx = c.getContext('2d');
 ctx.clearRect(0,0,c.width,c.height);

 let w = c.width / speeds.length;
 speeds.forEach((v,i)=>{
   ctx.fillRect(i*w, c.height - v*5, w-2, v*5);
 });
}

function generatePDF(){
 const { jsPDF } = window.jspdf;
 let doc = new jsPDF();

 let avg = document.getElementById('avg').innerText;
 let count = document.getElementById('count').innerText;

 doc.text("Airtel Service Report - Silchar", 10, 10);
 doc.text("Total Complaints: " + count, 10, 20);
 doc.text("Average Speed: " + avg, 10, 30);
 doc.text("Issue: Consistent degradation below promised speed.", 10, 40);

 doc.save("Airtel_Report.pdf");
}

// 🔄 REAL-TIME LISTENER
db.ref('reports').on('value', snap=>{
 speeds = [];
 let total = 0, count = 0;

 snap.forEach(c=>{
   let v = c.val().speed;
   speeds.push(v);
   total += v;
   count++;
 });

 document.getElementById('count').innerText = count;

 let avg = count ? (total/count).toFixed(2) : 0;
 document.getElementById('avg').innerText = avg + " Mbps";

 draw();
});
</script>
